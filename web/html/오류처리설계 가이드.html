🔧 오류 처리 데코레이터

반복적인 오류 처리를 위한 데코레이터를 만들어보겠습니다.

import functools
import logging

def error_handler(default_return=None, log_errors=True):
    """오류 처리 데코레이터"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            try:
                return func(*args, **kwargs)
            except Exception as e:
                if log_errors:
                    logging.error(f"{func.__name__}에서 오류 발생: {e}")
                return default_return
        return wrapper
    return decorator

# 사용 예시
@error_handler(default_return=False, log_errors=True)
def load_model(model_path):
    """모델 로딩 함수"""
    model = YOLO(model_path)
    return model

import os
from pathlib import Path

def safe_file_load(file_path):
    """안전한 파일 로딩"""
    try:
        if not os.path.exists(file_path):
            raise FileNotFoundError(f"파일을 찾을 수 없습니다: {file_path}")
        
        with open(file_path, 'r', encoding='utf-8') as f:
            return f.read()
    except FileNotFoundError as e:
        logger.error(f"파일 로딩 실패: {e}")
        return None
    except PermissionError:
        logger.error(f"파일 접근 권한이 없습니다: {file_path}")
        return None

ImportError / ModuleNotFoundError

모듈을 가져올 수 없을 때 발생

def check_dependencies():
    """필수 라이브러리 확인"""
    required_packages = {
        'cv2': 'opencv-python',
        'ultralytics': 'ultralytics',
        'numpy': 'numpy',
        'torch': 'torch'
    }
    
    missing_packages = []
    
    for package, install_name in required_packages.items():
        try:
            __import__(package)
            print(f"✅ {package} 설치됨")
        except ImportError:
            missing_packages.append(install_name)
            print(f"❌ {package} 설치 필요")
    
    if missing_packages:
        print(f"다음 패키지를 설치하세요: {', '.join(missing_packages)}")
        return False
    return True

from ultralytics import YOLO
import os

class YOLOModelManager:
    def __init__(self):
        self.model = None
        self.model_path = None
    
    def load_model(self, model_path):
        """안전한 YOLO 모델 로딩"""
        try:
            # 파일 존재 확인
            if not os.path.exists(model_path):
                raise FileNotFoundError(f"모델 파일을 찾을 수 없습니다: {model_path}")
            
            # 파일 확장자 확인
            if not model_path.endswith(('.pt', '.onnx', '.engine')):
                raise ValueError(f"지원하지 않는 모델 형식입니다: {model_path}")
            
            # 모델 로딩
            self.model = YOLO(model_path)
            self.model_path = model_path
            logger.info(f"✅ 모델 로딩 성공: {model_path}")
            return True
            
        except FileNotFoundError as e:
            logger.error(f"파일 오류: {e}")
            return False
        except Exception as e:
            logger.error(f"모델 로딩 실패: {e}")
            return False
    
    def predict_safe(self, image_path, conf=0.5):
        """안전한 예측 수행"""
        if self.model is None:
            logger.error("모델이 로딩되지 않았습니다.")
            return None
        
        try:
            results = self.model(image_path, conf=conf)
            return results
        except Exception as e:
            logger.error(f"예측 실패: {e}")
            return None

GPU 메모리 오류

import torch

def check_gpu_memory():
    """GPU 메모리 확인"""
    if torch.cuda.is_available():
        gpu_memory = torch.cuda.get_device_properties(0).total_memory
        gpu_memory_used = torch.cuda.memory_allocated(0)
        gpu_memory_free = gpu_memory - gpu_memory_used
        
        print(f"GPU 메모리 - 전체: {gpu_memory/1024**3:.2f}GB")
        print(f"GPU 메모리 - 사용: {gpu_memory_used/1024**3:.2f}GB")
        print(f"GPU 메모리 - 여유: {gpu_memory_free/1024**3:.2f}GB")
        
        if gpu_memory_free < 1024**3:  # 1GB 미만
            print("⚠️ GPU 메모리 부족!")
            return False
    return True

def process_with_gpu_check(model, image_batch):
    """GPU 메모리 체크와 함께 처리"""
    if not check_gpu_memory():
        # CPU로 전환
        model.to('cpu')
        print("CPU 모드로 전환됩니다.")
    
    try:
        results = model(image_batch)
        return results
    except torch.cuda.OutOfMemoryError:
        print("GPU 메모리 부족으로 CPU로 전환합니다.")
        torch.cuda.empty_cache()
        model.to('cpu')
        return model(image_batch)

    except Exception as e:
        logger.error(f"알 수 없는 오류: {e}")
        return None

웹캠 연결 오류

class SafeVideoCapture:
    def __init__(self, source=0):
        self.source = source
        self.cap = None
        self.is_opened = False
    
    def open(self):
        """안전한 비디오 캡처 열기"""
        try:
            self.cap = cv2.VideoCapture(self.source)
            
            if not self.cap.isOpened():
                raise RuntimeError(f"비디오 소스를 열 수 없습니다: {self.source}")
            
            # 테스트 프레임 읽기
            ret, frame = self.cap.read()
            if not ret:
                raise RuntimeError("비디오 프레임을 읽을 수 없습니다.")
            
            self.is_opened = True
            logger.info(f"✅ 비디오 캡처 열기 성공: {self.source}")
            return True
            
        except RuntimeError as e:
            logger.error(f"비디오 캡처 오류: {e}")
            return False
        except Exception as e:
            logger.error(f"알 수 없는 비디오 오류: {e}")
            return False
    
    def read_frame(self):
        """안전한 프레임 읽기"""
        if not self.is_opened or self.cap is None:
            return None
        
        try:
            ret, frame = self.cap.read()
            if not ret:
                logger.warning("프레임 읽기 실패")
                return None
            return frame
        except Exception as e:
            logger.error(f"프레임 읽기 오류: {e}")
            return None
    
    def release(self):
        """안전한 리소스 해제"""
        if self.cap is not None:
            self.cap.release()
            self.is_opened = False
            logger.info("비디오 캡처 해제됨")

🏆 모범 사례

완전한 오류 처리 템플릿

import logging
import cv2
from ultralytics import YOLO
import torch
import os
from pathlib import Path

# 로깅 설정
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('computer_vision.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

class ComputerVisionPipeline:
    def __init__(self):
        self.model = None
        self.device = self._get_device()
        
    def _get_device(self):
        """최적의 디바이스 선택"""
        if torch.cuda.is_available():
            logger.info("CUDA 사용 가능 - GPU 모드")
            return 'cuda'
        else:
            logger.info("CUDA 사용 불가 - CPU 모드")
            return 'cpu'
    
    def setup_model(self, model_path):
        """모델 설정"""
        try:
            # 파일 존재 확인
            if not Path(model_path).exists():
                raise FileNotFoundError(f"모델 파일 없음: {model_path}")
            
            # 모델 로딩
            self.model = YOLO(model_path)
            self.model.to(self.device)
            
            logger.info(f"✅ 모델 설정 완료: {model_path}")
            return True
            
        except Exception as e:
            logger.error(f"❌ 모델 설정 실패: {e}")
            return False
    
    def process_image(self, image_path, output_path=None):
        """이미지 처리"""
        try:
            # 전처리
            image = self._preprocess_image(image_path)
            if image is None:
                return None
            
            # 예측
            results = self._predict(image)
            if results is None:
                return None
            
            # 후처리
            output_image = self._postprocess(image, results)
            
            # 저장
            if output_path:
                self._save_result(output_image, output_path)
            
            return output_image
            
        except Exception as e:
            logger.error(f"이미지 처리 실패: {e}")
            return None
    
    def _preprocess_image(self, image_path):
        """이미지 전처리"""
        try:
            image = cv2.imread(image_path)
            if image is None:
                raise ValueError(f"이미지 로딩 실패: {image_path}")
            
            # 이미지 크기 확인
            height, width = image.shape[:2]
            if height == 0 or width == 0:
                raise ValueError(f"잘못된 이미지 크기: {image.shape}")
            
            return image
            
        except Exception as e:
            logger.error(f"이미지 전처리 실패: {e}")
            return None
    
    def _predict(self, image):
        """예측 수행"""
        try:
            if self.model is None:
                raise RuntimeError("모델이 로딩되지 않았습니다.")
            
            results = self.model(image)
            return results
            
        except torch.cuda.OutOfMemoryError:
            logger.warning("GPU 메모리 부족, CPU로 전환")
            self.model.to('cpu')
            return self.model(image)
        except Exception as e:
            logger.error(f"예측 실패: {e}")
            return None
    
    def _postprocess(self, image, results):
        """결과 후처리"""
        try:
            # 결과 그리기
            annotated_image = results[0].plot()
            return annotated_image
            
        except Exception as e:
            logger.error(f"후처리 실패: {e}")
            return image
    
    def _save_result(self, image, output_path):
        """결과 저장"""
        try:
            # 출력 디렉토리 생성
            output_dir = Path(output_path).parent
            output_dir.mkdir(parents=True, exist_ok=True)
            
            # 이미지 저장
            cv2.imwrite(str(output_path), image)
            logger.info(f"결과 저장: {output_path}")
            
        except Exception as e:
            logger.error(f"저장 실패: {e}")

if __name__ == "__main__":

    if pipeline.setup_model("yolov8n.pt"):
        result = pipeline.process_image("input.jpg", "output.jpg")
        if result is not None:
